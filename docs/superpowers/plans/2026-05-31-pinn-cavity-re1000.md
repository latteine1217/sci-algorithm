```python
"""設定資料結構與 YAML 載入；不在此寫任務流水帳。"""
from __future__ import annotations
from dataclasses import dataclass, field
import yaml
import jax


@dataclass
class NetworkConfig:
    width: int = 64
    depth: int = 4
    n_fourier: int = 64
    fourier_sigma: float = 2.0


@dataclass
class OptimizerConfig:
    learning_rate: float = 1.0e-3
    b1: float = 0.95
    b2: float = 0.95
    weight_decay: float = 0.0
    precondition_frequency: int = 10
    decay_steps: int = 20000


@dataclass
class TrainConfig:
    steps: int = 20000
    n_collocation: int = 16384
    resample_every: int = 100
    weight_update_every: int = 100
    log_every: int = 500


@dataclass
class Config:
    re: float = 1000.0
    lid_r: float = 10.0
    seed: int = 0
    weighting: str = "gradnorm"
    curriculum: list = field(default_factory=list)
    network: NetworkConfig = field(default_factory=NetworkConfig)
    optimizer: OptimizerConfig = field(default_factory=OptimizerConfig)
    train: TrainConfig = field(default_factory=TrainConfig)


def load_config(path: str) -> Config:
    with open(path, "r") as f:
        raw = yaml.safe_load(f)
    net = NetworkConfig(**raw.pop("network", {}))
    opt = OptimizerConfig(**raw.pop("optimizer", {}))
    tr = TrainConfig(**raw.pop("train", {}))
    return Config(network=net, optimizer=opt, train=tr, **raw)


def device_info() -> str:
    """回報 JAX 後端與裝置，供日誌記錄（Observability）。"""
    return f"backend={jax.default_backend()} devices={jax.devices()}"
```

```python
"""幾何：單位方腔的距離函數、lid 正則化剖面、collocation 取樣。

距離函數 D 在四壁為 0，用於 hard-BC 包裝。
lid 剖面採 cosh 型，角點為 0、中心≈1，避免角點速度不連續。
"""
import jax
import jax.numpy as jnp


def distance_fn(x, y):
    """D(x,y)=x(1-x)y(1-y)，四壁為 0、內部為正。"""
    return x * (1.0 - x) * y * (1.0 - y)


def lid_profile(x, r: float = 10.0):
    """g(x)=1 - cosh(r(x-0.5))/cosh(0.5 r)；g(0)=g(1)=0, g(0.5)=1-sech(0.5r)≈1。"""
    return 1.0 - jnp.cosh(r * (x - 0.5)) / jnp.cosh(0.5 * r)


def sample_collocation(key, n: int):
    """單位方腔內均勻取樣 n 個內部 collocation 點，形狀 (n,2)。"""
    return jax.random.uniform(key, shape=(n, 2), minval=0.0, maxval=1.0)
```

```python
"""網路：Random Fourier Features + Modified MLP（Wang et al. 2021）+ hard-BC 包裝。

參數以純 dict pytree 手寫，保持 jax.grad/jit/vmap 透明，利於後續演算法研究。
Fourier 矩陣 B 為固定（非訓練），以 stop_gradient 凍結。
"""
import jax
import jax.numpy as jnp
from .geometry import distance_fn, lid_profile


def _glorot(key, shape):
    fan_in, fan_out = shape[0], shape[1]
    scale = jnp.sqrt(2.0 / (fan_in + fan_out))
    return jax.random.normal(key, shape) * scale


def init_params(key, net_cfg):
    """初始化 Modified MLP 參數。輸入維度 = 2*n_fourier。"""
    width, depth, nf, sigma = (
        net_cfg.width, net_cfg.depth, net_cfg.n_fourier, net_cfg.fourier_sigma,
    )
    keys = jax.random.split(key, 6 + depth)
    in_dim = 2 * nf  # cos,sin 串接
    params = {}
    # 固定 Fourier 矩陣 B: (2, nf)
    params["B"] = jax.random.normal(keys[0], (2, nf)) * sigma
    # U/V 編碼器：in_dim -> width
    params["Wu"] = _glorot(keys[1], (in_dim, width)); params["bu"] = jnp.zeros(width)
    params["Wv"] = _glorot(keys[2], (in_dim, width)); params["bv"] = jnp.zeros(width)
    # 首層：in_dim -> width
    params["W0"] = _glorot(keys[3], (in_dim, width)); params["b0"] = jnp.zeros(width)
    # 隱藏層：width -> width，共 depth-1 層
    params["Wh"] = []; params["bh"] = []
    for i in range(depth - 1):
        params["Wh"].append(_glorot(keys[4 + i], (width, width)))
        params["bh"].append(jnp.zeros(width))
    # 輸出層：width -> 3
    params["Wout"] = _glorot(keys[-1], (width, 3)); params["bout"] = jnp.zeros(3)
    return params


def _fourier(params, xy):
    """xy: (...,2) -> (...,2*nf) cos/sin 特徵。B 凍結。"""
    B = jax.lax.stop_gradient(params["B"])
    proj = 2.0 * jnp.pi * (xy @ B)  # (...,nf)
    return jnp.concatenate([jnp.cos(proj), jnp.sin(proj)], axis=-1)


def forward(params, xy):
    """Modified MLP 原始輸出 (u_hat,v_hat,p_hat)，xy: (...,2) -> (...,3)。"""
    h_in = _fourier(params, xy)
    U = jnp.tanh(h_in @ params["Wu"] + params["bu"])
    V = jnp.tanh(h_in @ params["Wv"] + params["bv"])
    H = jnp.tanh(h_in @ params["W0"] + params["b0"])
    for Wh, bh in zip(params["Wh"], params["bh"]):
        Z = jnp.tanh(H @ Wh + bh)
        H = (1.0 - Z) * U + Z * V
    return H @ params["Wout"] + params["bout"]


def predict(params, xy, lid_r: float):
    """套 hard-BC 後的物理輸出 (u,v,p)，xy: (...,2) -> (...,3)。"""
    raw = forward(params, xy)
    x = xy[..., 0]; y = xy[..., 1]
    g = lid_profile(x, r=lid_r)
    D = distance_fn(x, y)
    u = y * g + D * raw[..., 0]
    v = D * raw[..., 1]
    p = raw[..., 2]
    return jnp.stack([u, v, p], axis=-1)
```

```python
"""穩態不可壓 NS 殘差（velocity-pressure）。

逐點以 autodiff 取一階/二階導數，再 vmap 批次化。
殘差：
  mom_x = u u_x + v u_y + p_x - (1/Re)(u_xx+u_yy)
  mom_y = u v_x + v v_y + p_y - (1/Re)(v_xx+v_yy)
  cont  = u_x + v_y
"""
import jax
import jax.numpy as jnp
from .networks import predict


def _pointwise_residual(params, xy_pt, re, lid_r):
    """xy_pt: (2,) -> (mom_x, mom_y, cont) 純量三元組。"""
    def f(pt):  # pt:(2,) -> (3,)
        return predict(params, pt[None, :], lid_r=lid_r)[0]

    val = f(xy_pt)                      # (3,)
    jac = jax.jacrev(f)(xy_pt)          # (3,2): d(u,v,p)/d(x,y)
    hess = jax.jacfwd(jax.jacrev(f))(xy_pt)  # (3,2,2)

    u, v, _ = val[0], val[1], val[2]
    u_x, u_y = jac[0, 0], jac[0, 1]
    v_x, v_y = jac[1, 0], jac[1, 1]
    p_x, p_y = jac[2, 0], jac[2, 1]
    u_xx, u_yy = hess[0, 0, 0], hess[0, 1, 1]
    v_xx, v_yy = hess[1, 0, 0], hess[1, 1, 1]

    nu = 1.0 / re
    mom_x = u * u_x + v * u_y + p_x - nu * (u_xx + u_yy)
    mom_y = u * v_x + v * v_y + p_y - nu * (v_xx + v_yy)
    cont = u_x + v_y
    return mom_x, mom_y, cont


def ns_residuals(params, xy, re: float, lid_r: float):
    """xy: (N,2) -> (rx, ry, rc) 各 (N,)。"""
    rx, ry, rc = jax.vmap(
        lambda pt: _pointwise_residual(params, pt, re, lid_r)
    )(xy)
    return rx, ry, rc
```

```python
"""損失組裝與自適應權重。

hard-BC 已免去 BC penalty，故僅平衡三項 PDE 殘差。
gradnorm（learning-rate annealing, Wang et al. 2021）：依各項對參數的梯度範數調權重。
"""
import jax
import jax.numpy as jnp
from .physics import ns_residuals


def loss_terms(params, xy, re, lid_r):
    """回傳 (Lx, Ly, Lc) 三項殘差 MSE 純量。"""
    rx, ry, rc = ns_residuals(params, xy, re=re, lid_r=lid_r)
    return jnp.mean(rx ** 2), jnp.mean(ry ** 2), jnp.mean(rc ** 2)


def init_weights():
    return {"x": 1.0, "y": 1.0, "c": 1.0}


def total_loss(params, xy, weights, re, lid_r):
    lx, ly, lc = loss_terms(params, xy, re=re, lid_r=lid_r)
    return weights["x"] * lx + weights["y"] * ly + weights["c"] * lc


def _grad_norm(loss_fn, params):
    g = jax.grad(loss_fn)(params)
    leaves = jax.tree_util.tree_leaves(g)
    sq = sum(jnp.sum(l ** 2) for l in leaves)
    return jnp.sqrt(sq) + 1e-12


def update_weights(params, xy, re, lid_r, method="gradnorm"):
    """回傳新權重 dict。gradnorm：lambda_i = mean_grad_norm / grad_norm_i。"""
    if method == "fixed":
        return init_weights()
    if method == "gradnorm":
        gx = _grad_norm(lambda p: loss_terms(p, xy, re, lid_r)[0], params)
        gy = _grad_norm(lambda p: loss_terms(p, xy, re, lid_r)[1], params)
        gc = _grad_norm(lambda p: loss_terms(p, xy, re, lid_r)[2], params)
        mean_g = (gx + gy + gc) / 3.0
        return {"x": float(mean_g / gx), "y": float(mean_g / gy), "c": float(mean_g / gc)}
    if method == "ntk":
        # NTK 對角近似：以各殘差對參數梯度的平方範數倒數為權重（簡化版）
        gx = _grad_norm(lambda p: loss_terms(p, xy, re, lid_r)[0], params) ** 2
        gy = _grad_norm(lambda p: loss_terms(p, xy, re, lid_r)[1], params) ** 2
        gc = _grad_norm(lambda p: loss_terms(p, xy, re, lid_r)[2], params) ** 2
        mean_g = (gx + gy + gc) / 3.0
        return {"x": float(mean_g / gx), "y": float(mean_g / gy), "c": float(mean_g / gc)}
    raise ValueError(f"unknown weighting method: {method}")
```

```python
"""訓練編排：單階段 SOAP 訓練迴圈 + 可選 Re curriculum + checkpoint。

Observability：定期記錄 loss 與各殘差項，便於回答「為何收斂/不收斂」。
"""
import pickle
import jax
import jax.numpy as jnp
import optax

from .networks import init_params, predict
from .losses import total_loss, loss_terms, update_weights, init_weights
from .optimizers import build_optimizer
from .geometry import sample_collocation
from .config import device_info


def _train_one_stage(params, cfg, re, key):
    """以 SOAP 在固定 re 訓練 cfg.train.steps 步，回傳 (params, history)。"""
    opt = build_optimizer(cfg.optimizer)
    state = opt.init(params)
    weights = init_weights()
    lid_r = cfg.lid_r
    history = {"loss": [], "lx": [], "ly": [], "lc": []}

    @jax.jit
    def step(params, state, xy, weights):
        L, grads = jax.value_and_grad(
            lambda p: total_loss(p, xy, weights, re, lid_r)
        )(params)
        updates, state = opt.update(grads, state, params)
        params = optax.apply_updates(params, updates)
        return params, state, L

    xy = sample_collocation(key, cfg.train.n_collocation)
    for it in range(cfg.train.steps):
        if it % cfg.train.resample_every == 0 and it > 0:
            key, sk = jax.random.split(key)
            xy = sample_collocation(sk, cfg.train.n_collocation)
        if it % cfg.train.weight_update_every == 0:
            weights = update_weights(params, xy, re, lid_r, method=cfg.weighting)
        params, state, L = step(params, state, xy, weights)
        if it % cfg.train.log_every == 0 or it == cfg.train.steps - 1:
            lx, ly, lc = loss_terms(params, xy, re, lid_r)
            history["loss"].append(float(L))
            history["lx"].append(float(lx)); history["ly"].append(float(ly)); history["lc"].append(float(lc))
            print(f"[Re={re:.0f}] it={it} loss={float(L):.3e} "
                  f"lx={float(lx):.2e} ly={float(ly):.2e} lc={float(lc):.2e}")
    return params, history


def train(cfg):
    """主入口。curriculum 為空 → 直攻 cfg.re；否則逐階段 warm-start。"""
    print(f"=== Train start | {device_info()} ===")
    key = jax.random.PRNGKey(cfg.seed)
    key, ik = jax.random.split(key)
    params = init_params(ik, cfg.network)

    stages = list(cfg.curriculum) if cfg.curriculum else [cfg.re]
    if stages[-1] != cfg.re:
        stages.append(cfg.re)

    full_history = {"loss": [], "lx": [], "ly": [], "lc": []}
    for re in stages:
        key, sk = jax.random.split(key)
        params, hist = _train_one_stage(params, cfg, float(re), sk)
        for k in full_history:
            full_history[k].extend(hist[k])
    return params, full_history


def save_checkpoint(params, path):
    with open(path, "wb") as f:
        pickle.dump(jax.device_get(params), f)


def load_checkpoint(path):
    with open(path, "rb") as f:
        return pickle.load(f)
```

```python
"""驗證：Ghia 中線比對、相對 L2、流線/場圖。

matplotlib 圖標題與 label 用英文（LANGUAGE_POLICY）。
"""
import numpy as np
import jax.numpy as jnp
from .networks import predict
from .benchmark_ghia import GHIA_RE1000


def relative_l2(pred, ref):
    pred = np.asarray(pred); ref = np.asarray(ref)
    return float(np.linalg.norm(pred - ref) / (np.linalg.norm(ref) + 1e-12))


def centerline_profiles(params, lid_r):
    """於 Ghia 取樣點預測中線 u/v，回傳 pred 與 ghia 對照。"""
    g = GHIA_RE1000
    ys = np.array(g["y"]); xs_v = np.array(g["x"])
    # 垂直中線 x=0.5：u(y)
    pts_u = jnp.stack([jnp.full_like(jnp.array(ys), 0.5), jnp.array(ys)], axis=-1)
    u_pred = np.array(predict(params, pts_u, lid_r=lid_r)[:, 0])
    # 水平中線 y=0.5：v(x)
    pts_v = jnp.stack([jnp.array(xs_v), jnp.full_like(jnp.array(xs_v), 0.5)], axis=-1)
    v_pred = np.array(predict(params, pts_v, lid_r=lid_r)[:, 1])
    return {
        "y": ys, "u_pred": u_pred, "u_ghia": np.array(g["u"]),
        "x": xs_v, "v_pred": v_pred, "v_ghia": np.array(g["v"]),
    }


def evaluate(params, lid_r, out_dir="results"):
    """完整驗證：印 L2、存中線對比圖與場圖。回傳指標 dict。"""

    os.makedirs(out_dir, exist_ok=True)

    prof = centerline_profiles(params, lid_r)
    l2_u = relative_l2(prof["u_pred"], prof["u_ghia"])
    l2_v = relative_l2(prof["v_pred"], prof["v_ghia"])
    print(f"=== Validation === rel-L2 u={l2_u:.4f}  v={l2_v:.4f}")

    # 中線剖面對比
    fig, ax = plt.subplots(1, 2, figsize=(10, 4))

    n = 100
    xs = np.linspace(0, 1, n); ys = np.linspace(0, 1, n)

    cf = ax.contourf(XX, YY, speed, levels=30, cmap="viridis")
    ax.streamplot(XX, YY, U, V, color="white", density=1.2, linewidth=0.6)
    ax.set_title("Velocity magnitude & streamlines (Re=1000)")
    ax.set_xlabel("x"); ax.set_ylabel("y"); fig.colorbar(cf, ax=ax)
    fig.tight_layout(); fig.savefig(f"{out_dir}/field.png", dpi=150); plt.close(fig)

    return {"rel_l2_u": l2_u, "rel_l2_v": l2_v}
```
